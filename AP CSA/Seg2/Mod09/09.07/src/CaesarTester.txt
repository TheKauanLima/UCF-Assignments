import java.util.Scanner;
public class CaesarTester
{
	public static void main(String[] args)
	{
		// create and declare necessary objects and variables
		Scanner in = new Scanner(System.in);
		int shiftValue;
		String decision1 = "";
		String decision2 = "";
		String plaintext = "";
		
		//repeat program until exit
		while (true)
		{
			// prompt for input value
			System.out.print("Please input a shift value (0-25): ");
			shiftValue = in.nextInt();
			System.out.println();
		
			// determine whether value is acceptable
			if (shiftValue < 0 || shiftValue > 25)
			{
			System.out.println("Invalid value");
			System.exit(0);
			}

			// prompt for selection
			System.out.println("Would you like to Encyrpt or Decrypt?");
			decision1 = in.next();
			System.out.println();
			
			// encrypt or decrypt
			if (decision1.toUpperCase().charAt(0) == 'E')
			{
				
				Encryption caesar = new Encryption(shiftValue);
				Encryption.cipher();
				System.out.println();
				
				System.out.println("Type message to encrypt: ");

				in.nextLine();
				plaintext = in.nextLine();
				caesar.encrypt(plaintext);
				System.out.println();
			}
			else if (decision1.toUpperCase().charAt(0) == 'D')
			{
				Decryption caesar = new Decryption(shiftValue);
				Decryption.cipher();
				System.out.println();
				
				System.out.println("Type message to decrypt: ");

				in.nextLine();
				plaintext = in.nextLine();
				caesar.decrypt(plaintext);
				System.out.println();
			}
			else
			{
				System.out.println("Invalid Response");
				System.exit(0);
			}
			
			// prompt for further encryption
			System.out.println("Would you like to cipher more? (y/n)");
			decision2 = in.next();
			System.out.println();
			
			// repeat or end program
			if (decision2.toUpperCase().charAt(0) == 'N')
			{
				System.out.println("Thank you for your time");
				System.exit(0);
			}
			else
			{
				System.out.println("Invalid Response");
				System.exit(0);
			}
		}
	}
}
